//学生表的连表查询
       SELECT
            t.*, f.advisorName
        FROM
            (
                SELECT
                    legao_student.*, schoolAreaName
                FROM
                    legao_student
                LEFT JOIN (
                    SELECT
                        id AS schoolAreaIDs,
                        NAME AS schoolAreaName
                    FROM
                        legao_schoolarea
                ) a ON a.schoolAreaIDs = legao_student.schoolAreaID
            ) t
        LEFT JOIN (
            SELECT
                id AS followID,
                advisor AS advisorName
            FROM
                legao_follow
        ) f ON f.followID = t.followID
        WHERE id = #{id};
